🔧 GuardianForegroundService — Kya karta hai?

  • START_STICKY — System agar service kill kare, Android ise automatically restart karta hai
  • PARTIAL_WAKE_LOCK — Screen off hone ke baad bhi CPU active rehta hai, SOS trigger ke time GPS/SMS chalta rehta hai
  • PRIORITY_MAX + FLAG_ONGOING_EVENT — User notification dismiss nahi kar sakta (non-dismissible)
  • WakeLock sirf SOS active hone par acquire hota hai, aur deactivate hone par release — battery safe
GuardianForegroundService.kt Kotlin 
package com.lunacare.app.service

import android.app.Notification
import android.app.NotificationChannel
import android.app.NotificationManager
import android.app.PendingIntent
import android.app.Service
import android.content.Context
import android.content.Intent
import android.os.Build
import android.os.IBinder
import android.os.PowerManager
import androidx.core.app.NotificationCompat
import com.lunacare.app.MainActivity
import com.lunacare.app.R

class GuardianForegroundService : Service() {

    // ── WakeLock reference (CPU ko jagaye rakhta hai)
    private var wakeLock: PowerManager.WakeLock? = null

    // ── Notification Channel ID
    private val CHANNEL_ID   = "lunacare_guardian_channel"
    private val NOTIFICATION_ID = 1001

    // ════════════════════════════════════════
    //  onCreate — Service pehli baar create
    // ════════════════════════════════════════
    override fun onCreate() {
        super.onCreate()
        createNotificationChannel()   // Android 8+ requires channel
        startForeground(
            NOTIFICATION_ID,
            buildGuardianNotification()   // Non-dismissible high-priority notification
        )
        acquireWakeLock()   // CPU lock — screen off hone pe bhi chalta rahe
    }

    // ════════════════════════════════════════
    //  onStartCommand — START_STICKY → system restart karega
    // ════════════════════════════════════════
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        // Handle SOS trigger / deactivate actions via Intent extras
        when (intent?.action) {
            ACTION_SOS_ACTIVATE   -> handleSosActivate()
            ACTION_SOS_DEACTIVATE -> handleSosDeactivate()
        }
        // START_STICKY: service kill hone ke baad Intent=null ke saath restart hota hai
        return START_STICKY
    }

    // ════════════════════════════════════════
    //  onBind — Bound service nahi hai
    // ════════════════════════════════════════
    override fun onBind(intent: Intent?): IBinder? = null

    // ════════════════════════════════════════
    //  onDestroy — WakeLock release karo
    // ════════════════════════════════════════
    override fun onDestroy() {
        releaseWakeLock()
        super.onDestroy()
    }

    // ════════════════════════════════════════
    //  SOS Activate — WakeLock acquire
    // ════════════════════════════════════════
    private fun handleSosActivate() {
        acquireWakeLock()
        // TODO: Start your GPS tracking / SMS loop here
        updateNotification(
            title   = "🆘 SOS ACTIVE — LunaCare Guardian",
            message = "Emergency mode active. Location updating every 4.5 min."
        )
    }

    // ════════════════════════════════════════
    //  SOS Deactivate — WakeLock release
    // ════════════════════════════════════════
    private fun handleSosDeactivate() {
        releaseWakeLock()
        // TODO: Stop GPS loop / SMS here
        updateNotification(
            title   = "🛡️ LunaCare Guardian Active",
            message = "Monitoring in background. Tap to open."
        )
    }

    // ════════════════════════════════════════
    //  WAKELOCK — PARTIAL_WAKE_LOCK
    //  Screen off ho, CPU chalta rahe
    // ════════════════════════════════════════
    private fun acquireWakeLock() {
        if (wakeLock?.isHeld == true) return   // Already holding, skip
        val pm = getSystemService(Context.POWER_SERVICE) as PowerManager
        wakeLock = pm.newWakeLock(
            PowerManager.PARTIAL_WAKE_LOCK,
            "LunaCare::GuardianWakeLock"
        ).apply {
            // Timeout: 30 minutes max to prevent indefinite drain
            // SOS active hone par re-acquire karna hoga
            acquire(30 * 60 * 1000L)
        }
    }

    private fun releaseWakeLock() {
        if (wakeLock?.isHeld == true) {
            wakeLock?.release()
        }
        wakeLock = null
    }

    // ════════════════════════════════════════
    //  NOTIFICATION — Non-dismissible, PRIORITY_MAX
    // ════════════════════════════════════════
    private fun buildGuardianNotification(
        title:   String = "🛡️ LunaCare Guardian Active",
        message: String = "Monitoring in background. Tap to open."
    ): Notification {
        // Tapping notification → open MainActivity
        val tapIntent = PendingIntent.getActivity(
            this,
            0,
            Intent(this, MainActivity::class.java).apply {
                flags = Intent.FLAG_ACTIVITY_SINGLE_TOP
            },
            PendingIntent.FLAG_UPDATE_CURRENT or PendingIntent.FLAG_IMMUTABLE
        )

        return NotificationCompat.Builder(this, CHANNEL_ID)
            .setContentTitle(title)
            .setContentText(message)
            .setSmallIcon(R.drawable.ic_luna_shield)   // Apna icon rakho
            .setContentIntent(tapIntent)
            .setPriority(NotificationCompat.PRIORITY_MAX)   // Highest priority
            .setCategory(NotificationCompat.CATEGORY_SERVICE)
            .setOngoing(true)           // Non-dismissible (FLAG_ONGOING_EVENT)
            .setShowWhen(false)         // No timestamp clutter
            .setAutoCancel(false)       // User dismiss nahi kar sakta
            .build()
    }

    private fun updateNotification(title: String, message: String) {
        val nm = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager
        nm.notify(NOTIFICATION_ID, buildGuardianNotification(title, message))
    }

    // ════════════════════════════════════════
    //  NOTIFICATION CHANNEL — Android 8+ required
    // ════════════════════════════════════════
    private fun createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val channel = NotificationChannel(
                CHANNEL_ID,
                "LunaCare Guardian Service",
                NotificationManager.IMPORTANCE_HIGH   // Heads-up notification
            ).apply {
                description     = "Keeps LunaCare SOS running in background"
                setShowBadge(false)
                lockscreenVisibility = Notification.VISIBILITY_PUBLIC
            }
            val nm = getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager
            nm.createNotificationChannel(channel)
        }
    }

    // ════════════════════════════════════════
    //  COMPANION OBJECT — Intent Actions &
    //  Helper to start/stop service safely
    // ════════════════════════════════════════
    companion object {
        const val ACTION_SOS_ACTIVATE   = "com.lunacare.SOS_ACTIVATE"
        const val ACTION_SOS_DEACTIVATE = "com.lunacare.SOS_DEACTIVATE"

        /** Call this from Activity/Fragment to start Guardian service */
        fun startGuardian(context: Context) {
            val intent = Intent(context, GuardianForegroundService::class.java)
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                context.startForegroundService(intent)   // Android 8+ requires this
            } else {
                context.startService(intent)
            }
        }

        /** Trigger SOS mode */
        fun triggerSOS(context: Context) {
            val intent = Intent(context, GuardianForegroundService::class.java)
                .apply { action = ACTION_SOS_ACTIVATE }
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                context.startForegroundService(intent)
            } else {
                context.startService(intent)
            }
        }

        /** Deactivate SOS mode */
        fun deactivateSOS(context: Context) {
            val intent = Intent(context, GuardianForegroundService::class.java)
                .apply { action = ACTION_SOS_DEACTIVATE }
            context.startService(intent)
        }
    }
}

🔄 BootReceiver — Kya karta hai?

  • ACTION_BOOT_COMPLETED — Phone reboot hone ke baad service automatically restart hoti hai
  • ACTION_MY_PACKAGE_REPLACED — App update install hone ke baad bhi service restart hoti hai (user ko manually open nahi karna padega)
  • Dono broadcast ek hi receiver handle karta hai — clean & minimal
  • Manifest mein android:directBootAware="true" lagao taaki encrypted boot pe bhi kaam kare (Android 7+)
BootReceiver.kt Kotlin 
package com.lunacare.app.receiver

import android.content.BroadcastReceiver
import android.content.Context
import android.content.Intent
import android.os.Build
import android.util.Log
import com.lunacare.app.service.GuardianForegroundService

class BootReceiver : BroadcastReceiver() {

    // ════════════════════════════════════════
    //  onReceive — Boot ya App-Replace broadcast
    // ════════════════════════════════════════
    override fun onReceive(context: Context, intent: Intent) {
        val action = intent.action ?: return
        Log.d("BootReceiver", "Received broadcast: $action")

        when (action) {
            // ── Phone reboot complete ──
            Intent.ACTION_BOOT_COMPLETED,
            // ── Encrypted boot complete (Android 7+) ──
            "android.intent.action.LOCKED_BOOT_COMPLETED",
            // ── App update installed ──
            Intent.ACTION_MY_PACKAGE_REPLACED -> {
                startGuardianService(context)
            }
        }
    }

    // ════════════════════════════════════════
    //  Start Guardian Foreground Service
    // ════════════════════════════════════════
    private fun startGuardianService(context: Context) {
        try {
            val serviceIntent = Intent(context, GuardianForegroundService::class.java)

            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
                // Android 8+ — startForegroundService() use karna ZAROORI hai
                // Warna Android "Context.startForegroundService() did not call
                // Service.startForeground()" ANR dega
                context.startForegroundService(serviceIntent)
            } else {
                context.startService(serviceIntent)
            }

            Log.d("BootReceiver", "GuardianForegroundService started successfully.")

        } catch (e: Exception) {
            // Background service restriction (Android 12+) ya permission missing
            Log.e("BootReceiver", "Failed to start service: ${e.message}")
        }
    }
}

🔋 BatteryOptimizationHelper — Kya karta hai?

  • Android ka Doze Mode background apps ko sleep kar deta hai — GPS & SMS ruk jaata hai
  • Ye helper user ko system settings mein le jaata hai jahan woh app ko battery optimization se exempt kar sakti hai
  • REQUEST_IGNORE_BATTERY_OPTIMIZATIONS permission se direct dialog bhi trigger ho sakta hai
  • Onboarding ya SOS setup ke waqt ye prompt dikhao — ek baar allow hone ke baad dobara nahi poochna padta
BatteryOptimizationHelper.kt Kotlin 
package com.lunacare.app.util

import android.annotation.SuppressLint
import android.app.Activity
import android.content.Context
import android.content.Intent
import android.net.Uri
import android.os.Build
import android.os.PowerManager
import android.provider.Settings
import androidx.appcompat.app.AlertDialog

object BatteryOptimizationHelper {

    // ════════════════════════════════════════
    //  Check: Kya app already exempt hai?
    // ════════════════════════════════════════
    fun isIgnoringBatteryOptimizations(context: Context): Boolean {
        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.M) return true  // Pre-M mein no Doze
        val pm = context.getSystemService(Context.POWER_SERVICE) as PowerManager
        return pm.isIgnoringBatteryOptimizations(context.packageName)
    }

    // ════════════════════════════════════════
    //  Method 1: Direct System Dialog
    //  (ACTION_REQUEST_IGNORE_BATTERY_OPTIMIZATIONS)
    //  Note: Manifest mein REQUEST_IGNORE_BATTERY_OPTIMIZATIONS
    //  permission hona ZAROORI hai
    // ════════════════════════════════════════
    @SuppressLint("BatteryLife")
    fun requestDirectExemption(activity: Activity) {
        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.M) return
        if (isIgnoringBatteryOptimizations(activity)) return  // Already exempt

        val intent = Intent(Settings.ACTION_REQUEST_IGNORE_BATTERY_OPTIMIZATIONS).apply {
            data = Uri.parse("package:${activity.packageName}")
        }
        try {
            activity.startActivity(intent)
        } catch (e: Exception) {
            // Kuch devices pe ye crash karta hai — fallback to settings page
            openBatterySettings(activity)
        }
    }

    // ════════════════════════════════════════
    //  Method 2: App-specific battery settings
    //  (Yeh sabse reliable hai — user manually toggle kare)
    // ════════════════════════════════════════
    fun openBatterySettings(activity: Activity) {
        val intent = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) {
            Intent(Settings.ACTION_IGNORE_BATTERY_OPTIMIZATION_SETTINGS)
        } else {
            Intent(Settings.ACTION_SETTINGS)
        }
        try {
            activity.startActivity(intent)
        } catch (e: Exception) {
            // Absolute fallback
            activity.startActivity(Intent(Settings.ACTION_SETTINGS))
        }
    }

    // ════════════════════════════════════════
    //  Show Explanation Dialog
    //  (User ko batao kyun zaroor hai — then ask)
    // ════════════════════════════════════════
    fun showBatteryOptimizationDialog(activity: Activity) {
        if (isIgnoringBatteryOptimizations(activity)) return  // Already done, skip

        AlertDialog.Builder(activity)
            .setTitle("🛡️ Battery Optimization — Disable karo")
            .setMessage(
                "LunaCare ka SOS sirf tab kaam karega jab app background mein " +
                "chal rahi ho.\n\n" +
                "Android ka Doze Mode app ko sleep kar deta hai, jiski wajah se " +
                "emergency mein GPS aur SMS ruk sakta hai.\n\n" +
                "Please next screen mein LunaCare ko 'Unrestricted' ya 'Don't optimize' " +
                "par set karo."
            )
            .setPositiveButton("Allow karo ✅") { _, _ ->
                requestDirectExemption(activity)
            }
            .setNegativeButton("Baad mein") { dialog, _ ->
                dialog.dismiss()
                // Warn user SOS may not work reliably
            }
            .setCancelable(false)  // Non-dismissible dialog
            .show()
    }

    // ════════════════════════════════════════
    //  Call this on SOS Setup or Onboarding
    //  Usage in Activity/Fragment:
    //
    //  BatteryOptimizationHelper.checkAndPrompt(this)
    // ════════════════════════════════════════
    fun checkAndPrompt(activity: Activity) {
        if (!isIgnoringBatteryOptimizations(activity)) {
            showBatteryOptimizationDialog(activity)
        }
    }
}

📡 LocationTelemetryService — Kya karta hai?

  • FusedLocationProviderClient — Google Play Services ka best-in-class GPS provider, jo GPS + WiFi + Cell Tower sab combine karta hai maximum accuracy ke liye
  • PRIORITY_HIGH_ACCURACY — Sabse zyada precise location milti hai, battery optimize hoti hai baaki time
  • 20-sec interval / 10-sec fastest interval — Regular updates ke saath urgent data bhi milta hai
  • Zero-Latency Injection — SOS trigger hone par turant ek immediate high-priority fix force hoti hai — standard interval wait nahi karna padta
  • Firebase Realtime Database — Har location update /emergency_logs/{uid} node pe push hoti hai with lat, lng, speed, accuracy, serverTimestamp
  • Location Disabled Handling — Agar GPS off hai to user ko High Accuracy GPS enable karne ka dialog dikhaata hai

 *
 *  Dependencies (app/build.gradle mein add karo):
 *  implementation "com.google.android.gms:play-services-location:21.0.1"
 *  implementation "com.google.firebase:firebase-database-ktx:20.3.0"
 *  implementation "com.google.firebase:firebase-auth-ktx:22.3.0"
 * ══════════════════════════════════════════════════════════════════
 */
class LocationTelemetryService(private val context: Context) {

    companion object {
        private const val TAG = "LocationTelemetry"

        // ── Configuration ──────────────────────────────────────────
        private const val UPDATE_INTERVAL_MS      = 20_000L   // 20 seconds — normal interval
        private const val FASTEST_INTERVAL_MS     = 10_000L   // 10 seconds — fastest allowed
        private const val SOS_IMMEDIATE_TIMEOUT   = 5_000L    // 5 sec max wait for instant fix

        // ── Firebase DB Path ───────────────────────────────────────
        private const val DB_PATH = "emergency_logs"
    }

    // ── FusedLocationProviderClient — Google ka best GPS provider ──
    private val fusedClient: FusedLocationProviderClient =
        LocationServices.getFusedLocationProviderClient(context)

    // ── Firebase References ────────────────────────────────────────
    private val firebaseAuth = FirebaseAuth.getInstance()
    private val firebaseDb   = FirebaseDatabase.getInstance()

    // ── LocationRequest — PRIORITY_HIGH_ACCURACY ───────────────────
    // Yeh config regular SOS tracking ke liye hai
    private val locationRequest: LocationRequest = LocationRequest.Builder(
        Priority.PRIORITY_HIGH_ACCURACY,
        UPDATE_INTERVAL_MS
    ).apply {
        setMinUpdateIntervalMillis(FASTEST_INTERVAL_MS)   // Fastest allowed interval: 10 sec
        setWaitForAccurateLocation(true)                  // Pehli reading accurate hone tak wait karo
        setMaxUpdateDelayMillis(UPDATE_INTERVAL_MS * 2)   // Max batch delay
    }.build()

    // ── Location Callback — Har update par Firebase push ──────────
    private val locationCallback = object : LocationCallback() {

        override fun onLocationResult(result: LocationResult) {
            result.lastLocation?.let { location ->

                // ── Speed: m/s → km/h convert karo ─────────────────
                val speedKmh = if (location.hasSpeed()) {
                    (location.speed * 3.6f).toDouble()   // 1 m/s = 3.6 km/h
                } else {
                    0.0
                }

                // ── Telemetry data object ───────────────────────────
                val telemetryData = mapOf(
                    "latitude"        to location.latitude,
                    "longitude"       to location.longitude,
                    "speed_kmh"       to speedKmh,
                    "accuracy_meters" to location.accuracy.toDouble(),
                    "altitude"        to location.altitude,
                    "bearing"         to location.bearing.toDouble(),
                    "provider"        to (location.provider ?: "unknown"),
                    "serverTimestamp" to ServerValue.TIMESTAMP,    // Firebase server time — tamper-proof
                    "deviceTimestamp" to location.time             // Device ka local time (backup)
                )

                pushToFirebase(telemetryData)

                Log.d(TAG, "📍 Location pushed: ${location.latitude}, ${location.longitude} | " +
                    "Speed: ${"%.1f".format(speedKmh)} km/h | " +
                    "Accuracy: ${location.accuracy}m")
            }
        }

        override fun onLocationAvailability(availability: LocationAvailability) {
            if (!availability.isLocationAvailable) {
                Log.w(TAG, "⚠️ Location unavailable — GPS disabled ya signal nahi hai")
                // UI thread pe GPS disabled dialog dikhao
                android.os.Handler(Looper.getMainLooper()).post {
                    showLocationDisabledDialog()
                }
            }
        }
    }

    // ════════════════════════════════════════════════════════════════
    //  startTracking() — Regular 20-sec SOS tracking shuru karo
    // ════════════════════════════════════════════════════════════════
    fun startTracking() {
        // Pehle check karo GPS enabled hai ya nahi
        if (!isLocationEnabled()) {
            showLocationDisabledDialog()
            return
        }

        try {
            fusedClient.requestLocationUpdates(
                locationRequest,
                locationCallback,
                Looper.getMainLooper()   // Main thread pe callback (UI safe)
            )
            Log.d(TAG, "✅ Location tracking started — 20sec interval, PRIORITY_HIGH_ACCURACY")
        } catch (e: SecurityException) {
            Log.e(TAG, "❌ Location permission missing: ${e.message}")
        }
    }

    // ════════════════════════════════════════════════════════════════
    //  stopTracking() — Tracking band karo (SOS deactivate par)
    // ════════════════════════════════════════════════════════════════
    fun stopTracking() {
        fusedClient.removeLocationUpdates(locationCallback)
        Log.d(TAG, "🛑 Location tracking stopped")
    }

    // ════════════════════════════════════════════════════════════════
    //  forceImmediateFix() — ZERO-LATENCY INJECTION
    //  ─────────────────────────────────────────────────────────────
    //  SOS trigger hone ke TURANT baad ye call karo.
    //  Standard 20-sec interval ka wait bypass karta hai aur
    //  ek immediate high-priority location fix force karta hai.
    //
    //  Usage:
    //      telemetryService.forceImmediateFix()   // SOS button press ke turant baad
    //      telemetryService.startTracking()        // Regular loop start karo
    // ════════════════════════════════════════════════════════════════
    fun forceImmediateFix() {
        // ── One-shot high-priority immediate request ────────────────
        val immediateRequest = LocationRequest.Builder(
            Priority.PRIORITY_HIGH_ACCURACY,
            1L   // Interval = 1ms → turant deliver karo
        ).apply {
            setMaxUpdates(1)                           // Sirf EK fix chahiye
            setWaitForAccurateLocation(false)          // Don't wait — speed priority hai
            setMinUpdateIntervalMillis(1L)
        }.build()

        try {
            fusedClient.requestLocationUpdates(
                immediateRequest,
                object : LocationCallback() {
                    override fun onLocationResult(result: LocationResult) {
                        result.lastLocation?.let { location ->

                            val speedKmh = if (location.hasSpeed()) {
                                (location.speed * 3.6f).toDouble()
                            } else { 0.0 }

                            // ── Immediate SOS telemetry with sosTriggered flag ──
                            val immediateData = mapOf(
                                "latitude"        to location.latitude,
                                "longitude"       to location.longitude,
                                "speed_kmh"       to speedKmh,
                                "accuracy_meters" to location.accuracy.toDouble(),
                                "altitude"        to location.altitude,
                                "serverTimestamp" to ServerValue.TIMESTAMP,
                                "deviceTimestamp" to location.time,
                                "sosTriggered"    to true,   // Ye SOS trigger fix hai — flag lagao
                                "provider"        to (location.provider ?: "unknown")
                            )
                            pushToFirebase(immediateData)
                            // Ek baar push hone ke baad ye callback hata do
                            fusedClient.removeLocationUpdates(this)

                            Log.d(TAG, "🚨 ZERO-LATENCY SOS Fix: ${location.latitude}, " +
                                "${location.longitude} | Accuracy: ${location.accuracy}m")
                        }
                    }
                },
                Looper.getMainLooper()
            )
            Log.d(TAG, "⚡ Immediate high-priority location fix requested")
        } catch (e: SecurityException) {
            Log.e(TAG, "❌ Permission missing for immediate fix: ${e.message}")
        }
    }

    // ════════════════════════════════════════════════════════════════
    //  pushToFirebase() — /emergency_logs/{uid} pe data push karo
    // ════════════════════════════════════════════════════════════════
    private fun pushToFirebase(data: Map<String, Any>) {
        val uid = firebaseAuth.currentUser?.uid ?: run {
            // Auth nahi hai — anonymous UID use karo ya local device ID
            Log.w(TAG, "⚠️ No Firebase user — using anonymous node")
            "anonymous_${android.os.Build.SERIAL}"
        }

        // push() → automatic unique key generate karta hai (timestamp-based)
        // Isse entries kabhi overwrite nahi hoti — har fix alag node mein jaati hai
        firebaseDb.reference
            .child(DB_PATH)           // /emergency_logs
            .child(uid)               // /emergency_logs/{uid}
            .push()                   // /emergency_logs/{uid}/{auto-key}
            .setValue(data)
            .addOnSuccessListener {
                Log.d(TAG, "✅ Firebase push successful")
            }
            .addOnFailureListener { err ->
                Log.e(TAG, "❌ Firebase push failed: ${err.message}")
                // TODO: Offline queue mein store karo aur baad mein retry karo
            }
    }

    // ════════════════════════════════════════════════════════════════
    //  isLocationEnabled() — GPS/High Accuracy enabled hai ya nahi
    // ════════════════════════════════════════════════════════════════
    fun isLocationEnabled(): Boolean {
        val lm = context.getSystemService(Context.LOCATION_SERVICE) as LocationManager
        return lm.isProviderEnabled(LocationManager.GPS_PROVIDER) ||
               lm.isProviderEnabled(LocationManager.NETWORK_PROVIDER)
    }

    // ════════════════════════════════════════════════════════════════
    //  showLocationDisabledDialog() — User ko GPS enable karne bolo
    //  High-Accuracy mode ka prompt dikhao
    // ════════════════════════════════════════════════════════════════
    private fun showLocationDisabledDialog() {
        // Ye sirf Activity context mein kaam karta hai
        // Service mein use kar rahe ho to Application context pe AlertDialog nahi banta
        // Recommended: LocalBroadcastManager se Activity ko message bhejo
        try {
            AlertDialog.Builder(context)
                .setTitle("📍 GPS Band Hai — Location Enable Karo")
                .setMessage(
                    "SOS emergency mein accurate location send karne ke liye\n\n" +
                    "Settings → Location → Mode → 'High Accuracy' enable karo.\n\n" +
                    "GPS disabled hone par contacts ko sirf approximate area milega — " +
                    "exact location ke liye High Accuracy GPS zaroori hai. 🛡️"
                )
                .setPositiveButton("Settings Kholo ✅") { _, _ ->
                    // High Accuracy Location Settings direct open karo
                    val intent = Intent(Settings.ACTION_LOCATION_SOURCE_SETTINGS).apply {
                        flags = Intent.FLAG_ACTIVITY_NEW_TASK
                    }
                    try {
                        context.startActivity(intent)
                    } catch (e: Exception) {
                        // Fallback: General settings
                        context.startActivity(
                            Intent(Settings.ACTION_SETTINGS).apply {
                                flags = Intent.FLAG_ACTIVITY_NEW_TASK
                            }
                        )
                    }
                }
                .setNegativeButton("Baad Mein") { dialog, _ ->
                    dialog.dismiss()
                    Log.w(TAG, "User ne GPS enable karna skip kiya — SOS accuracy reduced")
                }
                .setCancelable(false)   // Dismiss nahi kar sakta — safety critical
                .show()
        } catch (e: Exception) {
            // Window token error (service context) — broadcast se handle karo
            Log.e(TAG, "Dialog show failed (use Activity context): ${e.message}")
            // Send local broadcast to Activity to show dialog
            val broadcastIntent = Intent("com.lunacare.SHOW_GPS_DIALOG")
            context.sendBroadcast(broadcastIntent)
        }
    }
}

📋 AndroidManifest.xml — Telemetry ke liye extra permissions

AndroidManifest.xml — Telemetry additions XML 
<!-- ══ LOCATION TELEMETRY PERMISSIONS ══ -->

<!-- Fine + Coarse location (runtime permission bhi maango) -->
<uses-permission android:name="android.permission.ACCESS_FINE_LOCATION" />
<uses-permission android:name="android.permission.ACCESS_COARSE_LOCATION" />

<!-- Background location — SOS active hone par screen off pe bhi GPS -->
<!-- Android 10+ pe alag rationale dikhana padega user ko -->
<uses-permission android:name="android.permission.ACCESS_BACKGROUND_LOCATION" />

<!-- Internet — Firebase Realtime Database sync ke liye -->
<uses-permission android:name="android.permission.INTERNET" />

<!-- Network State — online/offline check ke liye -->
<uses-permission android:name="android.permission.ACCESS_NETWORK_STATE" />

<!-- Foreground service location type — Android 10+ mandatory -->
<service
    android:name=".service.GuardianForegroundService"
    android:enabled="true"
    android:exported="false"
    android:foregroundServiceType="location"
    android:stopWithTask="false" />

🔒 Firebase Realtime Database Rules — Security setup karo

Firebase Console → Realtime Database → Rules tab mein paste karo. Ye rules ensure karta hai ki sirf authenticated user apna data likh sakta hai aur admin (Emergency Responder) read kar sakta hai.

Firebase Realtime Database Rules JSON 
{
  "rules": {
    "emergency_logs": {
      "$uid": {
        // Sirf woh user likh sakta hai jiska uid match kare
        ".write": "auth != null && auth.uid === $uid",
        // Read: sirf authenticated users (emergency responders)
        ".read":  "auth != null"
      }
    }
  }
}
